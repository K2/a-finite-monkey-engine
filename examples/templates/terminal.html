import asyncio

# Ensure we have a proper event loop
try:
    loop = asyncio.get_event_loop()
except RuntimeError:
    loop = asyncio.new_event_loop()
    asyncio.set_event_loop(loop)

from finite_monkey.agents.orchestrator import WorkflowOrchestrator
